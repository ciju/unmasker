class window.SelectionRect
    constructor: (@x, @y, @w, @h) ->
        @div = $('<div class="sel-rect">')
        @rects = []
        $('body').append @div
        Selections.push @
        @id = GetId()
        @.updateDiv()
        @.setupEvents()
        @

    updateDiv: =>
        ns = @.NormalizedBoundary()
        @div.css(
            top: ns.y+'px'
            left: ns.x + 'px'
            width: ns.w + 'px'
            height: ns.h + 'px'
        )

    setupEvents: =>
        console.warn 'setting click'
        @div.click (e) ->
            if not Dragging
                e.preventDefault()
                e.stopPropagation()
        @div.dblclick (e) =>
            e.preventDefault()
            e.stopPropagation()

            $('.sel-rect').css 'display', 'none'
            $('.mask').css 'opacity', 1

        span = null
        @div.hover ((e) =>
            log 'hover'
            span = $('<span class="close"><a href="javascript:;">X</a></span>')
            @div.append span
        ), ((e) =>
            if span
                span.remove()
                span = null
        )

        @div.on 'click', '.close', (e) =>
            console.warn 'close clicked'
            @.Remove()
            e.stopPropagation()

    teardownEvents: ->
        @div.off 'click dblclick hover'

    Remove: ->
        @.teardownEvents()
        @div.remove()
        log 'selections ', Selections
        for s, i in Selections
            continue if not s
            log 'id ', @id, ' s.id', s.id
            continue if s.id != @id
            log 'remove the entry'
            Selections[i] = null
            $.publish 'selector.removed'
